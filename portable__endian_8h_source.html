105 // libcaer: define macros for floating-point conversions too.
106 // Floats must be 4 bytes in size, so like a 32-bit integer, but their endianness
107 // is undefined, so we use the 32bit conversion functions and memcpy() to do the
108 // conversions safely (assumed integer and floating-point do have the same
109 // endianness on the system, which is almost always true). For further details, see:
110 // https://stackoverflow.com/questions/10620601/portable-serialisation-of-ieee754-floating-point-values
111 static inline float htobeflt(float val) {
112  uint32_t rep;
113  memcpy(&rep, &val, sizeof(rep));
114  rep = htobe32(rep);
115  memcpy(&val, &rep, sizeof(rep));
116  return (val);
117 }
